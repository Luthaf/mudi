fn flat_iter(&self) -> Iter<S::Item>

Flat (linear) iteration over array elements.

use mudi::Array;
let mut array = Array::from_element(0, (3, 2));
for i in 0..3 {
    for j in 0..2 {
        array[(i, j)] = i;
    }
}

for value in array.flat_iter() {
    print!("{} ", value);
}
// This will print "0 0 1 1 2 2"
